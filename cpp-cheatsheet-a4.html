Arrays & Containers

Arrays

// Static arrays int arr[5] = {1, 2, 3, 4, 5}; int matrix[3][3] = {{1,2,3}, {4,5,6}, {7,8,9}}; // Access elements arr[0] = 10; // first element arr[4] = 50; // last element // Array size int size = sizeof(arr) / sizeof(arr[0]);

STL Containers

#include <vector> #include <map> #include <set> // Vector (dynamic array) vector<int> v = {1, 2, 3}; v.push_back(4); v.pop_back(); v[0] = 10; v.size(); // Map (key-value pairs) map<string, int> m; m["key"] = 42; m.insert({"hello", 1}); // Set (unique elements) set<int> s = {3, 1, 2}; s.insert(4); s.erase(1);

Functions

// Function declaration int add(int a, int b); // Function definition int add(int a, int b) { return a + b; } // Function with default parameters void greet(string name = "World") { cout << "Hello " << name << endl; } // Function overloading int multiply(int a, int b) { return a * b; } double multiply(double a, double b) { return a * b; } // Pass by reference void swap(int& a, int& b) { int temp = a; a = b; b = temp; } // Lambda functions (C++11) auto lambda = [](int x) { return x * 2; }; int result = lambda(5); // result = 10

Pointers & References

// Pointers int x = 42; int* ptr = &x; // pointer to x int value = *ptr; // dereference pointer // Null pointer int* nullPtr = nullptr; // C++11 int* oldNull = NULL; // older style // References int& ref = x; // reference to x ref = 100; // x is now 100 // Pointer arithmetic int arr[] = {1, 2, 3}; int* p = arr; p++; // points to arr[1] *(p + 1); // value of arr[2] // Dynamic memory int* dynamic = new int(42); delete dynamic; int* array = new int[10]; delete[] array;

Classes & Objects

class Rectangle { private: double width, height; public: // Constructor Rectangle(double w, double h) : width(w), height(h) {} // Default constructor Rectangle() : width(0), height(0) {} // Destructor ~Rectangle() { /* cleanup code */ } // Member functions double area() const { return width * height; } void setWidth(double w) { width = w; } double getWidth() const { return width; } // Static member static int count; static void printCount() { cout << count << endl; } }; // Usage Rectangle r(5.0, 3.0); cout << r.area() << endl;

Inheritance & Polymorphism

// Base class class Animal { protected: string name; public: Animal(string n) : name(n) {} virtual void makeSound() = 0; // pure virtual virtual ~Animal() {} // virtual destructor }; // Derived class class Dog : public Animal { public: Dog(string n) : Animal(n) {} void makeSound() override { cout << name << " barks!" << endl; } }; // Multiple inheritance class Pet { public: virtual void play() = 0; }; class Puppy : public Dog, public Pet { public: Puppy(string n) : Dog(n) {} void play() override { cout << "Playing fetch!" << endl; } };

Memory Management

// Smart pointers (C++11) #include <memory> // Unique pointer unique_ptr<int> uptr = make_unique<int>(42); int value = *uptr; // Shared pointer shared_ptr<int> sptr1 = make_shared<int>(42); shared_ptr<int> sptr2 = sptr1; // shared ownership cout << sptr1.use_count() << endl; // 2 // Weak pointer weak_ptr<int> wptr = sptr1; // RAII (Resource Acquisition Is Initialization) class FileHandler { FILE* file; public: FileHandler(const char* name) { file = fopen(name, "r"); } ~FileHandler() { if (file) fclose(file); } };

Templates

// Function template template<typename T> T maximum(T a, T b) { return (a > b) ? a : b; } // Usage int maxInt = maximum(5, 10); double maxDouble = maximum(3.14, 2.71); // Class template template<typename T> class Stack { private: vector<T> elements; public: void push(const T& item) { elements.push_back(item); } T pop() { T top = elements.back(); elements.pop_back(); return top; } }; // Template specialization template<> class Stack<bool> { // specialized implementation };

STL Algorithms

#include <algorithm> vector<int> v = {3, 1, 4, 1, 5}; // Sorting sort(v.begin(), v.end()); sort(v.rbegin(), v.rend()); // reverse sort // Searching auto it = find(v.begin(), v.end(), 4); bool found = (it != v.end()); // Binary search (sorted container) bool exists = binary_search(v.begin(), v.end(), 3); // Min/Max auto minIt = min_element(v.begin(), v.end()); auto maxIt = max_element(v.begin(), v.end()); // Count int count = count_if(v.begin(), v.end(), [](int x) { return x > 2; }); // Transform transform(v.begin(), v.end(), v.begin(), [](int x) { return x * 2; });

Exception Handling

// Try-catch block try { int result = 10 / 0; // may throw throw runtime_error("Custom error"); } catch (const runtime_error& e) { cout << "Runtime error: " << e.what() << endl; } catch (const exception& e) { cout << "General error: " << e.what() << endl; } catch (...) { cout << "Unknown error" << endl; } // Custom exception class MyException : public exception { public: const char* what() const noexcept override { return "My custom exception"; } }; // RAII with exceptions class Resource { public: Resource() { /* acquire resource */ } ~Resource() { /* release resource */ } };

Network Programming

Socket Basics (Unix/Linux)

#include <sys/socket.h> #include <netinet/in.h> #include <arpa/inet.h> #include <unistd.h> // TCP Server int server_fd = socket(AF_INET, SOCK_STREAM, 0); struct sockaddr_in address; address.sin_family = AF_INET; address.sin_addr.s_addr = INADDR_ANY; address.sin_port = htons(8080); bind(server_fd, (struct sockaddr*)&address, sizeof(address)); listen(server_fd, 3); int client_socket = accept(server_fd, NULL, NULL); char buffer[1024]; recv(client_socket, buffer, sizeof(buffer), 0); send(client_socket, "HTTP/1.1 200 OK\r\n\r\nHello", 25, 0); close(client_socket); close(server_fd);

TCP Client

// TCP Client int sock = socket(AF_INET, SOCK_STREAM, 0); struct sockaddr_in serv_addr; serv_addr.sin_family = AF_INET; serv_addr.sin_port = htons(8080); inet_pton(AF_INET, "127.0.0.1", &serv_addr.sin_addr); connect(sock, (struct sockaddr*)&serv_addr, sizeof(serv_addr)); send(sock, "Hello Server", 12, 0); char buffer[1024]; recv(sock, buffer, sizeof(buffer), 0); close(sock); // UDP Socket int udp_sock = socket(AF_INET, SOCK_DGRAM, 0); sendto(udp_sock, "data", 4, 0, (struct sockaddr*)&serv_addr, sizeof(serv_addr)); recvfrom(udp_sock, buffer, sizeof(buffer), 0, NULL, NULL); close(udp_sock);
